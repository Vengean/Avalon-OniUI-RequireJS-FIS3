                avalon.define("test", function(vm) {

                    vm.simplegrid = {
                        tbodyRowHeight: 105,

                            var lastyear = compare.lastyear

                                    return "<div>" + val + "</div><div class='datecell'>" + lastyear[name] + "<span>同比</span></div><div class='datecell'>" + lastday[name] + "<span>环比</span></div>"

                            }
                        },
