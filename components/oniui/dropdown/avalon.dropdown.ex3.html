multiple select（多选下拉框）

使用option配置dropdown并使用双工绑定

{{s2}}

    <!DOCTYPE html>
    <html>
    <head>
        <title></title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <script src="../avalon.js"></script>
        <script type="text/javascript">
         require(["dropdown/avalon.dropdown.js", "ready!"], function() {
                //multiple select vmodel
                avalon.define('mulSelect', function(vm) {
                    vm.$skipArray = ['opts'];
                    vm.opts = {
                        data: [
                            {
                                value: '1',
                                label: '1'
                            }, {
                                value: 2,
                                label: '2'
                            }, {
                                label: 'group1',
                                options: [
                                    {
                                        value: false,
                                        label: 'false'
                                    },
                                    {
                                        value: null,
                                        label: 'null'
                                    }
                                ]
                            }, {
                                label: 'group2',
                                options: [
                                    {
                                        value: '3',
                                        label: '3'
                                    }
                                ]
                            }
                        ],
                        multiple: true,
                        autofocus: true,
                        size: 4,
                        enable: false,
                        modleBind: true
                    };
                    vm.s2 = ['1', 2, null, false, '3'];
                });
                avalon.scan();
            });
        </script>
    </head>
    <body>
    <div ms-controller="mulSelect">
        <h2>multiple select（多选下拉框）</h2>
        <p>使用option配置dropdown并使用双工绑定</p>
        <select ms-widget="dropdown,$,opts" ms-duplex="s2"></select>
        <p>{{s2}}</p>
    </div>
    </body>
    </html>
