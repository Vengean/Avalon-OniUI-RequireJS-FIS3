multiple select（多选下拉框）

使用html配置dropdown并附加双工绑定，默认处理value的值为Number，Boolean，null的情况

select1: {{s1}}

<!DOCTYPE html>
<html>
<head>
    <title></title>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width">
    <script src="../avalon.js"></script>
    <script type="text/javascript">
            require(["dropdown/avalon.dropdown.js", "ready!"], function() {
                //multiple select vmodel
                avalon.define('mulSelect', function(vm) {
                    vm.s1 = ['value1', 1, null, false];
                });
                avalon.scan();
            });
    </script>
</head>
<body>
    <div class="wrapper">
        <div ms-controller="mulSelect">
            <h2>multiple select（多选下拉框）</h2>
            <p>使用html配置dropdown并附加双工绑定，默认处理value的值为Number，Boolean，null的情况</p>
            <select ms-duplex="s1"
                    ms-widget="dropdown"
                    multiple>
                <optgroup label="group1">
                    <option value="value1">Value 1</option>
                    <option>textValue</option>
                    <option label="label" value="labelValue">with label</option>
                    <option value="null">Value 2</option>
                    <option value="1">Value 3</option>
                </optgroup>
                <optgroup label="group2">
                    <option value="false">Value 4</option>
                    <option value="value5" disabled>Value 5</option>
                    <option value="value6">Value 6</option>
                </optgroup>
            </select>
            <p>select1: {{s1}}</p>
        </div>
    </div>
</body>
</html>
