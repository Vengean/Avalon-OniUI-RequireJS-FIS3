multiple select（多选下拉框）

用select结构直接生成dropdown multiple组件

        <!DOCTYPE html>
        <html>
        <head>
            <title></title>
            <meta charset="UTF-8">
            <meta name="viewport" content="width=device-width">
            <script src="../avalon.js"></script>
            <script type="text/javascript">
                require('dropdown/avalon.dropdown.js', function() {
                    avalon.ready(function() {
                        //multiple select vmodel
                        avalon.define('mulSelect', function(vm) {});
                        avalon.scan();
                    });
                });
            </script>
        </head>
        <body>
            <div class="wrapper">
                <div ms-controller="mulSelect">
                    <h1>multiple select（多选下拉框）</h1>
                    <p>用select结构直接生成dropdown multiple组件</p>
                    <select ms-widget="dropdown"
                            multiple>
                        <optgroup label="group1">
                            <option value="value1">Value 1</option>
                            <option>textValue</option>
                            <option label="label" value="labelValue">with label</option>
                            <option value="value2" selected>Value 2</option>
                            <option value="value3">Value 3</option>
                        </optgroup>
                        <optgroup label="group2">
                            <option value="value4" selected>Value 4</option>
                            <option value="value5" disabled selected>Value 5</option>
                            <option value="value6">Value 6</option>
                        </optgroup>
                    </select>
                </div>
            </div>
        </body>
        </html>
