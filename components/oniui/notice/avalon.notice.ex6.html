notice demo

配置notice type为success，设置successClass来配置其样式类，同理当type为error时可以配置errorClass，当type为默认info时可以配置infoClass

                <!DOCTYPE HTML>
                <html>
                <head>
                    <meta charset='utf-8'/>
                    <title>notice demo 1</title>
                    <script src='avalon.js'></script>
                </head>
                <body>
                    <div class="demo-show" ms-controller="noticeDemo">
                        <div ms-widget="notice,a" data-notice-type="success">配置notice type为success，设置successClass来配置其样式类，同理当type为error时可以配置errorClass，当type为默认info时可以配置infoClass</div>
                        <button ms-click="toggleNotice('a')">toggle notice a</button>
                        <button ms-click="changeTypeClass('a')">change class of notice a</button>
                        <script>
                            require(["notice/avalon.notice"], function(avalon) {
                                var model = avalon.define("noticeDemo", function(vm) {
                                    vm.toggleNotice = function(id) {
                                        var noticeVm = avalon.vmodels[id]
                                        noticeVm.toggle = !noticeVm.toggle
                                    }
                                    vm.changeTypeClass = function(id) {
                                        var noticeVm = avalon.vmodels[id]
                                        noticeVm.successClass = "oni-notice-success"
                                    }
                                })
                                avalon.scan();
                            })
                        </script>
                    </div>
                </body>
                </html>
