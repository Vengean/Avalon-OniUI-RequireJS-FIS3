left {{$index}} right {{$index}} 

<!DOCTYPE html>
<html>
    <head>
        <title>scrollbar组件</title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <script src="../avalon.js"></script>
        
        <script>
            require(["scrollbar/avalon.scrollbar"], function() {
                avalon.define("test", function(vm) {
                    vm.scrollbar = {
                        size: "large"
                    }
                    vm.data = avalon.range(1,100)
                    vm.$index = 2
                    vm.$skipArray = ["scrollbar"]
                    vm.up = function() {
                        avalon.vmodels.$hehe.update()
                    }
                })
                avalon.scan()
            })
        </script>
        <style type="text/css">
            .nobarbg .oni-scrollbar{
                background: transparent;
            }
        </style>
    </head>
    <body ms-controller="test" style="padding-bottom:85000px;">
        <h1>scrollbar-元素带上下小箭头双滚动条</h1>
        <div ms-widget="scrollbar,$hehe" class="oni-helper-reset" style="height:300px;border:1px solid #ccc;width:700px;" data-scrollbar-position="right,bottom">
            <div class="oni-scrollbar-scroller oni-helper-reset">
                <p style="width:65000px" ms-repeat="data" class="oni-helper-reset" data-repeat-rendered="up">left {{$index}} <span style="float:right;">right {{$index}} </span></p>
            </div>
        </div>
        
    </div>
</body>
