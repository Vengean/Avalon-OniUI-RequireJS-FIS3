<!DOCTYPE html>
<html>
    <head>
        <title>scrollbar组件</title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <script src="../avalon.js"></script>
        
        <script>
            require(["scrollbar/avalon.scrollbar"], function() {
                avalon.define("test", function(vm) {
                    vm.scrollbar = {
                        size: "small"
                    }
                    vm.data = avalon.range(1,400)
                    vm.$index = 2
                    vm.$skipArray = ["scrollbar"]
                    vm.up = function() {
                        avalon.vmodels.$cc.update()
                        avalon.vmodels.$dd.update()
                    }
                })
                avalon.scan()
            })
        </script>
        <style type="text/css">
            .nobarbg .oni-scrollbar{
                background: transparent;
            }
        </style>
    </head>
    <body>
<div ms-controller="test">
        <h1>scrollbar-带上下小箭头嵌套</h1>

            <div class="oni-scrollbar-scroller oni-helper-reset">
                <div ms-widget="scrollbar,$cc" class="oni-helper-reset" style="height:400px;border:1px solid #ccc;width:300px;overflow:hidden;" data-scrollbar-position="right,bottom">
                    <div class="oni-scrollbar-scroller oni-helper-reset">
                        <p style="width:800px" data-repeat-rendered="up" ms-repeat="data" class="oni-helper-reset">left {{$index}} <span style="float:right;">right {{$index}} </span></p>
                    </div>
                </div>
            </div>
        </div>
        
    </div>
