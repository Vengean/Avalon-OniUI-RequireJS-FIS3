非监控属性

在VM中，改变它们不会引起视图改变的属性，这包括以$开头的属性，其名字放在$skipArray中的属性，函数。

<!DOCTYPE html>
<html>
    <head>
        <title>unobservable</title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <script src="avalon.js"></script>
        <script>
            avalon.define("test", function(vm) {
                vm.$aaa = "sss"
                vm.$skipArray = ["bbb", "ccc"]
                vm.aaa = 1111
                vm.bbb = 222
                vm.ccc = 333
                vm.fn = function() {
                    return "函数"
                }
                vm.click = function(){
                    vm.$aaa = vm.aaa = vm.bbb = vm.ccc = "change"
                    vm.fn = function(){
                        return "----------"
                    }
                }
            })
           
        </script>
    </head>
    <body ms-controller="test">
        <p>{{$aaa}}</p>
        <p>{{aaa}}</p>
        <p>{{bbb}}</p>
        <p>{{ccc}}</p>
        <p>{{fn()}}</p>
        <button type="button" ms-click="click">点我</button>
    </body>
</html>
