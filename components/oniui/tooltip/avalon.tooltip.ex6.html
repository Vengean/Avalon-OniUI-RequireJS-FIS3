input元素获取焦点显示tip

<!DOCTYPE html>
<html>
    <head>
        <title>tooltip组件</title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <script src="../avalon.js"></script>
        
        <style type="text/css">
            body{
                padding-bottom: 400px;
            }
            div.ct{
                padding: 40px 0;
            }
            a{
                font-size: 13px;
                margin-left: 20px;
                display: inline-block;
            }
        </style>
        <script>
            var str = '请输入';
            require(["tooltip/avalon.tooltip"], function() {
                vmodel = avalon.define("test", function(vm) {
                    vm.$ppOpt = {
                        "contentGetter": function(elem) {
                            if(elem.tagName.toLowerCase() == 'input' || elem.getAttribute("tp")) return str
                        }
                        , autohide: true
                        , event: 'focus'
                    }
                    vm.$skipArray = ["tooltip"]
                })
                avalon.scan()
            })
        </script>
    </head>
    <body class="oni-helper-reset">
<div ms-controller="test">
        <h1>input元素获取焦点显示tip</h1>
        <div class="ct">
            <input type="text" ms-widget="tooltip,pp, $ppOpt" data-tooltip-delegate="false" data-tooltip-position="tr" data-tooltip-position-my="left bottom-10" data-tooltip-position-at="right-20 top" value="focus me"/>
        </div>
        
    </div>
</body>
</html>
