移动端时间选择器

                <!DOCTYPE HTML>
                <html>
                <head>
                    <meta charset='utf-8'/>
                    <title>timer demo</title>
                    <script src='avalon.js'></script>
                </head>
                <body ms-controller="test">
                    <h1>移动端时间选择器</h1>
                    <input ms-widget="timer" data-timer-event-type="mousewheel"/>
                    <button ms-click="getCurrentTime">get current time</button>
                    <script>
                        require(["timer/avalon.timer"], function() {
                            var timerVM = null
                            avalon.define("test", function(vm) {
                                vm.timer = {
                                    onInit: function(vmodel, options, vmodels) {
                                        timerVM = vmodel
                                    }
                                }
                                vm.getCurrentTime = function() {
                                    var currentDate = timerVM.timeDate
                                    if (currentDate) {
                                        alert("当前选中日期为："+currentDate.toLocaleString())
                                    } else {
                                        alert("还未选择时间，请选择")
                                    }
                                }
                            })
                            avalon.scan()
                        })
                    </script>
                </body>
                </html>
