textbox demo 7

自适应高度，组件标签必须为textarea，可配置最大最小高度

不配置最大最小高度，可无限延伸，默认最小高度为60px

                <div class="wrapper">
                    <h1>textbox demo 7</h1>

                    <p class="example">自适应高度，组件标签必须为textarea，可配置最大最小高度</p>
                    <div class="demo-show" ms-controller="demo7">
                        <textarea ms-widget="textbox, textbox, $textboxOpts1" ms-duplex="content1" cols="50" placeholder="尝试不断复制上面的文字进来"></textarea>
                    </div>

                    <p class="example">不配置最大最小高度，可无限延伸，默认最小高度为60px</p>
                    <div class="demo-show" ms-controller="demo7">
                        <textarea ms-widget="textbox, textbox, $textboxOpts2" ms-duplex="content2" cols="50" placeholder="尝试不断复制上面的文字进来"></textarea>
                    </div>

                    <script>
                        require(["textbox/avalon.textbox"], function() {
                            avalon.define("demo7", function(vm) {
                                vm.$textboxOpts1 = {
                                    adaptiveHeight: {
                                        minHeight: 30,
                                        maxHeight: 200
                                    }
                                }
                                vm.$textboxOpts2 = {
                                    adaptiveHeight: {
                                        minHeight: null, //不定义自适应范围
                                        maxHeight: null
                                    }
                                }
                                vm.content1 = ""
                                vm.content2 = ""
                            })
                            avalon.scan();
                        })
                    </script>
                </div>
