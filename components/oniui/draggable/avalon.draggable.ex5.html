        <script>
            require(["draggable/avalon.draggable"], function() {
                avalon.define("test", function(vm) {
                    vm.draggable = {
                        handle: function(e){
                            var el = e.target
                            do{
                                if(el.className === "handle"){
                                    return el
                                }
                            }while(el = el.parentNode)
                        }
                    }
                    vm.$skipArray = ["draggable"]
                })
                avalon.scan()
            })

        </script>
