改变dialog的z-index（zIndex=300）

说明：通过改变dialog vmodel的zIndex属性来改变dialog组件的z-index

                <!DOCTYPE HTML>
                <html>
                <head>
                    <meta charset='utf-8'/>
                    <title>最基本的dialog组件</title>
                    <script src='avalon.js'></script>
                </head>
                <body>
                    <div class="demo-show" ms-controller="demo">
                        <fieldset>  
                            <legend>改变dialog的z-index</legend>
                                <button ms-click="show('aa')">打开对话框</button>
                                <div ms-widget="dialog,aa">
                                    呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵
                                    <p>{{message}}</p>
                                </div>
                                <button ms-click="changeZIndex('aa')">change z-index of dialog 'aa'</button>
                        </fieldset>
                        <script> 
                            require(["dialog/avalon.dialog"], function(avalon) {
                                avalon.define("demo", function(vm) {
                                    vm.message = "笑傻了吧！！";
                                    vm.show = function( id ){
                                        avalon.vmodels[id].toggle = true;
                                    }
                                    vm.changeZIndex = function(id) {
                                        avalon.vmodels[id].zIndex = 300;
                                    }
                                })
                                avalon.scan();
                            })
                        </script>
                    </div>
                </body>
                </html>
