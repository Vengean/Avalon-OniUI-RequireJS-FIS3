不显示关闭按钮的dialog

说明：此demo基本使用组件默认配置，只是通过showClose的配置使得组件不显示“关闭”按钮

                <!DOCTYPE HTML>
                <html>
                <head>
                    <meta charset='utf-8'/>
                    <title>不显示关闭按钮的dialog（</title>
                    <script src='avalon.js'></script>
                </head>
                <body>
                    <div class="demo-show" ms-controller="demo">
                        <fieldset>  
                            <legend>取消“close”按钮的dialog</legend>
                                <button ms-click="show('aa')">打开对话框</button>
                                <div ms-widget="dialog,aa,$aaOpts">
                                    呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵
                                </div>
                        </fieldset>
                        <script> 
                            require(["dialog/avalon.dialog"], function(avalon) {
                                avalon.define("demo", function(vm) {
                                    vm.$aaOpts = {
                                        title: "取消“close”按钮的dialog",
                                        showClose: false
                                    }
                                    vm.show = function( id ){
                                        avalon.vmodels[id].toggle = true;
                                    }
                                })
                                avalon.scan();
                            })
                        </script>
                    </div>
                </body>
                </html>
