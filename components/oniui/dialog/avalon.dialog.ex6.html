模拟alert（type="alert"）

说明：当配置type为alert时，showClose的配置无效，也就是其为false还是true都不会显示“关闭”按钮

                <!DOCTYPE HTML>
                <html>
                <head>
                    <meta charset='utf-8'/>
                    <title>模拟alert</title>
                    <script src='avalon.js'></script>
                </head>
                <body>
                    <div class="demo-show" ms-controller="demo">
                        <fieldset>  
                            <legend>show-close在type为alert情况下配置无效</legend>
                                <button ms-click="show('bb')">打开对话框</button>
                                <div ms-widget="dialog,bb,opts"> 
                                    设置了type之后，无论showClose为true or false都不显示关闭Icon
                                </div>
                        </fieldset>
                        <script> 
                            require(["dialog/avalon.dialog"], function(avalon) {
                                avalon.define("demo", function(vm) {
                                    vm.opts = {
                                        type: 'alert',
                                        title: '配置type为alert模拟alert',
                                        showClose: true
                                    };
                                    vm.show = function( id ){
                                        avalon.vmodels[id].toggle = true;
                                    }
                                })
                                avalon.scan();
                            })
                        </script>
                    </div>
                </body>
                </html>
