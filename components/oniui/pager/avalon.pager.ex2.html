指定回调onJump(与例6对照看)

<!DOCTYPE html>
<html>
    <head>
        <title>分页栏组件</title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <script src="../avalon.js">

        </script>
        <script>
            require(["pager/avalon.pager"], function() {
                avalon.define("test", function(vm) {
                    vm.pager = {
                        totalItems: 150,
                        onJump: function(e, data) {
                            location.href = "#page=" + data.currentPage
                        }
                    }
                    vm.$skipArray = ["pager"]
                })
                avalon.scan()
            })
        </script>
    </head>
    <body ms-controller="test">
        <h1>指定回调onJump</h1>
        <div ms-widget="pager"></div>

    </body>
</html>
