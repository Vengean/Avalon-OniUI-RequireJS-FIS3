<html>
    <head>
        <title>flipswitch组件</title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <style type="text/css">
            
        </style>
        <script src="../avalon.js"></script>
        
        <script>
            require(["flipswitch/avalon.flipswitch", "tooltip/avalon.tooltip"], function() {
                var id = 0;
                var vmodel = avalon.define("test", function(vm) {
                    vm["flipswitch"] = {
                        getTemplate: function(tpl) {
                            id++;
                            return tpl.replace(/^<div/, "<div ms-class='checked:checked' ")
                        }
                        , onChange: function(v) {
                            if(avalon.vmodels.$b2) {
                                avalon.vmodels.$b2.content = "单击" + (v ? "关闭" : "打开")
                            }
                        }
                    }
                    vm.$ttOpt = {
                        contentGetter: function(elem) {
                            var elem = avalon(elem)
                            if(elem.hasClass("oni-flipswitch")) return "单击" + (elem.hasClass("checked") ? "关闭" : "打开")
                        },
                        position: "tc"
                        , delegate: true,
                        autohide: false
                    }
                    vm.$skipArray = ["flipswitch"]
                })
                avalon.scan()
            })
        </script>
    </head>
    <body>
<div ms-controller="test">
        <h1>flipswitch + tooltip</h1>
        <div ms-widget="tooltip,$b2,$ttOpt">
            <p>---------size=normal---------</p>
            <div class="this-for-test-4">
                <input rank="3" checked="checked" type="checkbox" ms-widget="flipswitch,$dd" data-flipswitch-size="normal">
                <input type="button" value="disable" onclick="avalon.vmodels.$dd.disable()">
                <input type="button" value="enable" onclick="avalon.vmodels.$dd.enable()">
            </div>
            <p>---------size=large---------</p>
            <div class="this-for-test-2">
                <input rank="4" checked="checked" type="checkbox" ms-widget="flipswitch" data-flipswitch-size="large">
            </div>

        </div>
