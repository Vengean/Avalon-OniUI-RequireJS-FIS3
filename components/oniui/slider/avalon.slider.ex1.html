slider demo

基本的slider，step默认为1，min默认为0，max默认为100，默认水平拖动(orientation为“horizontal”)，range默认为false，这里配置了onDragStart、onDrag、onDragEnd回调，回调的参数都是对应事件的event对象和拖动的数据信息data，请打开控制台看看回调中打印的参数

                    <!DOCTYPE HTML>
                    <html>
                    <head>
                        <meta charset='utf-8'/>
                        <title>slider demo 1</title>
                        <script src='avalon.js'></script>
                    </head>
                    <body>
                        <div class="demo-show" ms-controller = "theme">
                            <div style="width:70%">
                                <input ms-widget="slider,aaa, $aaaOpts" >
                                <p ms-controller="aaa">值为{{value}}，百分比为{{percent}}%; min为{{$valueMin}}, max为{{$valueMax}}</p>
                            </div>
                            <script>
                                require(["slider/avalon.slider"], function() {
                                    var model = avalon.define("theme", function(vm) {
                                        vm.$aaaOpts = {
                                            onDragStart: function(event, data){
                                                avalon.log("start dragging");
                                                avalon.log(event);
                                                avalon.log(data);
                                            },
                                            onDrag: function(event, data) {
                                                avalon.log("dragging");
                                                avalon.log(event);
                                                avalon.log(data);
                                            },
                                            onDragEnd: function(data) {
                                                avalon.log("dragend");
                                                avalon.log(data);
                                            }
                                        }
                                    })
                                    avalon.scan()
                                })
                            </script>
                        </div>
                    </body>
                    </html>

值为{{value}}，百分比为{{percent}}%; min为{{$valueMin}}, max为{{$valueMax}}
