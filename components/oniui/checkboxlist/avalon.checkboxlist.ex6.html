checkboxlist demo 6

可以直接配置data属性来渲染组件， 当然也可以先什么也不配置，等取得数据的合适时机再动态的设置

注意： 当初始配置了data且同时设置了fetch，并且绑定元素列出了需要渲染的checkbox时，组件处理的优先级是：data > fetch > sub element

初始配置data
    {{ val }}
    延迟配置data

      {{ v1 }}

                      <!DOCTYPE HTML>
                <html>
                <head>
                    <meta charset='utf-8'/>
                    <title>checkboxlist demo6 </title>
                    <script src='avalon.js'></script>
                </head>
                <body>
                    <div class="demo-show" ms-controller="demo5">
                        <fieldset>
                            <legend>初始配置data</legend>
                            <ul ms-widget="checkboxlist, $, dataConfig" data-checkboxlist-duplex="val" data-checkboxlist-alltext="全部"></ul>
                            {{ val }}
                        </fieldset>
                        <fieldset>
                            <legend>延迟配置data</legend>

                            <ul ms-widget="checkboxlist, c1" data-checkboxlist-duplex="v1" data-checkboxlist-alltext="全选"></ul>
                            <p>{{ v1 }}</p>
                            <button ms-click="setData">set data</button>
                        </fieldset>
                        
                    </div>
                    <script>
                        require(["./checkboxlist/avalon.checkboxlist"], function(avalon) {
                            var sss = avalon.define('demo5',function(vm) {
                                vm.$skipArray = ["dataConfig"]
                                vm.val = ["1", "2"]
                                vm.v1 = []
                                vm.name = "coffee"
                                vm.dataConfig = {
                                    data: [{
                                        text: "选项一", 
                                        value: "1"
                                    }, {
                                        text: "选项二", 
                                        value: "2"
                                    }, {
                                        text: "选项三", 
                                        value: "3"
                                    }, {
                                        text: "选项四", 
                                        value: "4"
                                    }, {
                                        text: "选项五", 
                                        value: "5"
                                    }]
                                }
                                vm.setData = function() {
                                    avalon.vmodels.c1.data = [{
                                        text: "menu一{{name}}", 
                                        value: "11"
                                    }, {
                                        text: "menu二", 
                                        value: "22"
                                    }, {
                                        text: "menu三", 
                                        value: "33"
                                    }, {
                                        text: "menu四", 
                                        value: "44"
                                    }, {
                                        text: "menu五", 
                                        value: "55"
                                    }]
                                }
                            });
                            avalon.scan();
                        });
                    </script>
                </body>
                </html>
