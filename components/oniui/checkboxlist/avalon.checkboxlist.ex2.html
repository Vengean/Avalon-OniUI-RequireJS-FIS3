checkboxlist demo 2

配置checkboxlist-duplex初始化初始选中的选项，而且可以通过duplex值的改变修正选中项状态（data-checkboxlist-duplex="val"）

                <!DOCTYPE HTML>
                <html>
                <head>
                    <meta charset='utf-8'/>
                    <title>配置duplex选项</title>
                    <script src='avalon.js'></script>
                </head>
                <body>
                    <div class="demo-show" ms-controller="demo2">
                        <fieldset>
                            <legend>data-checkboxlist-duplex="val"</legend>
                            <ul ms-widget="checkboxlist" data-checkboxlist-duplex="val" data-checkboxlist-alltext="全部">
                                <li><input type="checkbox" value="1">选项一</li>
                                <li><input type="checkbox" value="2">选项二</li>
                                <li><input type="checkbox" value="3">选项三</li>
                                <li><input type="checkbox" value="4">选项四</li>
                                <li><input type="checkbox" value="5">选项五</li>
                                <li><input type="checkbox" value="6">选项六</li>
                            </ul>
                            {{ val }}
                        </fieldset>
                    </div>
                    <script>
                        require(["./checkboxlist/avalon.checkboxlist"], function(avalon) {
                            var sss = avalon.define('demo2',function(vm){
                                vm.val = ["1", "2"];
                            });
                            setTimeout(function (){
                                sss.val = ["1","2","3","4","5","6"];
                            },5000);
                            avalon.scan();
                        });
                    </script>
                </body>
                </html>
