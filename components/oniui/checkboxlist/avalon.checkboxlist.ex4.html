checkboxlist demo 4

配置checkboxlist-fetch获取用户定义的所有选项值（data-checkboxlist-fetch="get_options"）

                <!DOCTYPE HTML>
                <html>
                <head>
                    <meta charset='utf-8'/>
                    <title>配置duplex选项</title>
                    <script src='avalon.js'></script>
                </head>
                <body>
                    <div class="demo-show" ms-controller="demo4">
                        <fieldset>
                            <legend>data-checkboxlist-duplex="val"</legend>
                            <ul ms-widget="checkboxlist" data-checkboxlist-duplex="val" data-checkboxlist-alltext="全部" data-checkboxlist-fetch="get_options">
                            </ul>
                            {{ val }}
                        </fieldset>
                    </div>
                    <script>
                        require(["./checkboxlist/avalon.checkboxlist"], function(avalon) {
                            var sss = avalon.define('demo4',function(vm){
                                vm.val = ["w1", "w2"];
                                vm.get_options = function( callback ){
                                    setTimeout(function(){
                                        sss.val = ["w1","w2","w3"];
                                        callback([
                                            { text : '文字1' , value : 'w1' } ,
                                            { text : '文字2' , value : 'w2' } ,
                                            { text : '文字3' , value : 'w3' } ,
                                            { text : '文字4' , value : 'w4' }
                                        ]);
                                    },1000);
                                }
                            });
                            avalon.scan();
                        });
                    </script>
                </body>
                </html>
