checkboxlist demo 3

checkboxlist组件默认提供了type为week时的data（data-checkboxlist-type="week"）,只需简单配置就可完成星期一到星期日的选项的展示。如要配置初始选中项，值必须是["TUESDAY","MONDAY","WEDNESDAY","THURSDAY","FRIDAY","SATURDAY","SUNDAY"]中的表示星期几的英文全写

                <!DOCTYPE HTML>
                <html>
                <head>
                    <meta charset='utf-8'/>
                    <title>配置type为week</title>
                    <script src='avalon.js'></script>
                </head>
                <body>
                    <div class="demo-show" ms-controller="demo3">
                        <fieldse>
                            <legend>设置type为week展示周一到周日的选项</legend>
                            <ul ms-widget="checkboxlist" data-checkboxlist-type="week" data-checkboxlist-duplex="val" data-checkboxlist-alltext="全部">
                            </ul>
                        </fieldset>

                        {{ val }}
                    </div>
                    <script>
                        require(["./checkboxlist/avalon.checkboxlist"], function(avalon) {
                            avalon.define('demo3',function(vm){
                                vm.val = ['TUESDAY'];
                            });
                            avalon.scan();
                        });
                    </script>
                </body>
                </html>
