                <!DOCTYPE HTML>
                <html>
                <head>
                    <meta charset='utf-8'/>
                    <title>daterangepicker demo</title>
                    <script src='avalon.js'></script>
                </head>
                <body>
                    <div class="demo-show" ms-controller="demo">
                        <fieldset>
                            <legend>日期范围框 配置日期的显示规则和解析规则</legend>
                            <div ms-widget="daterangepicker, a, $aOpts"></div>
                        </fieldset>
                        <fieldset>
                            <legend>配置日期选择快捷按钮</legend>
                            <div ms-widget="daterangepicker" data-daterangepicker-shortcut="true"></div>
                        </fieldset>
                        <script> 
                            require(["datepicker/avalon.daterangepicker"], function(avalon) {
                                var model = avalon.define("demo", function(vm) {
                                    vm.$aOpts = {
                                        parseDate : function(str){
                                            var reg = "^(\\d{4})" + "年"+ "(\\d{1,2})"+ "月" +"(\\d{1,2})"+ "日$";
                                            reg = new RegExp(reg);
                                            var x = str.match(reg);
                                            return x ? new Date(x[1],x[2] * 1 -1 , x[3]) : null;
                                        },
                                        formatDate : function(date){
                                            if (!date) return
                                            var separator = this.separator,
                                                year = date.getFullYear(),
                                                month = date.getMonth(),
                                                day = date.getDate();
                                            return year + "年" + this.formatNum(month + 1 , 2) + "月" + this.formatNum(day , 2) + "日";
                                        }
                                    }
                                })
                                avalon.scan();
                            })
                        </script>
                    </div>
                </body>
                </html>
