datepicker lang demo

使用en语言包 
                <!DOCTYPE HTML>
                <html>
                <head>
                    <meta charset='utf-8'/>
                    <title>datepicker demo</title>
                    <script src='avalon.js'></script>
                </head>
                <body>
                    <h1>datepicker lang demo</h1>
                    <div class="demo-show" ms-controller="demo">
                        <fieldset style="position: relative; z-index: 2">
                            <legend>使用en语言包</legend>
                            <input ms-widget="datepicker"/>
                        </fieldset>
                        <script>
                            //将语言包加载即可
                            require(["datepicker/avalon.datepicker"], function(avalon) {
                                require(["i18n/avalon.datepicker.lang-en"], function() {
                                    var model = avalon.define("demo", function(vm) {
                                        //设置业务部分的文案
                                        vm.datepicker = {
                                            calendarLabel: "Datepicker",
                                            formatErrorTip: "Malformed"
                                        }
                                    })
                                    avalon.scan();
                                })
                            })
                        </script>
                    </div>
                </body>
                </html>
