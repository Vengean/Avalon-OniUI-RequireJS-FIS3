datepicker demo

toggle切换日期的显示隐藏

说明：可以通过toggle切换datepicker的显示与隐藏

calendarLabel配置日历的说明label

说明：配置calendarLabel来显示你独特的日历说明

                <!DOCTYPE HTML>
                <html>
                <head>
                    <meta charset='utf-8'/>
                    <title>datepicker demo</title>
                    <script src='avalon.js'></script>
                </head>
                <body>
                    <div class="demo-show" ms-controller="demo">
                        <fieldset style="position: relative; z-index: 2">
                            <legend>toggle切换日期的显示隐藏</legend>
                            <input ms-widget="datepicker, a"/>
                            <button ms-click="toggleDatepicker" style="float:right;">toggle calendar</button>
                            <p>说明：可以通过toggle切换datepicker的显示与隐藏</p>
                        </fieldset>
                        <fieldset style="position: relative; z-index: 1">
                            <legend>calendarLabel配置日历的说明label</legend>
                            <input ms-widget="datepicker" data-datepicker-calendar-label="选择你的日期" />
                            <p>说明：配置calendarLabel来显示你独特的日历说明</p>
                        </fieldset>
                        <script> 
                            require(["datepicker/avalon.datepicker"], function(avalon) {
                                var model = avalon.define("demo", function(vm) {
                                    vm.toggleDatepicker = function(event) {
                                        avalon.vmodels.a.toggle = !avalon.vmodels.a.toggle;
                                        event.stopPropagation();
                                    }
                                })
                                avalon.scan();
                            })
                        </script>
                    </div>
                </body>
                </html>
