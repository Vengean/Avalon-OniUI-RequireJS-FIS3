datepicker demo

带格式化输出配置

说明：配置dateCellRender用来格式化输出日期

                <!DOCTYPE HTML>
                <html>
                <head>
                    <meta charset='utf-8'/>
                    <title>datepicker demo</title>
                    <script src='avalon.js'></script>
                </head>
                <body>
                    <div class="demo-show" ms-controller="demo">
                        <fieldset style="position: relative; z-index: 2">
                            <legend>带格式化输出配置</legend>
                            <input ms-widget="datepicker"/>
                            <p>说明：配置dateCellRender用来格式化输出日期</p>
                        </fieldset>
                        <script> 
                            require(["datepicker/avalon.datepicker"], function(avalon) {
                                var model = avalon.define("demo", function(vm) {
                                    vm.datepicker = {
                                        dateCellRender: function(date, vmodel, dateObject) {
                                            if (date == 1) {
                                                return "<b style='color:green'>one</b>"
                                            }
                                            return date
                                        }
                                    }
                                })
                                avalon.scan();
                            })
                        </script>
                    </div>
                </body>
                </html>
