dropdownlist（下拉框）

通过duplex的设置获取选中值或者通过调用getSelected方法来得到选中值

<!DOCTYPE html>
<html>
    <head>
        <script src="../avalon.js"></script>
        <script type="text/javascript">
            require(["dropdownlist/avalon.dropdownlist", "ready!"], function() {
                avalon.define('dropdown', function(vm) {
                    vm.d2 = 'textValue'
                    vm.getAValue = function() {
                        console.log('调用getSelected方法获得组件A的选项值：' + avalon.vmodels.a.getSelected())
                    }
                    vm.getBValue = function() {
                        console.log('调用getSelected方法获得组件B的选项值: ' + avalon.vmodels.b.getSelected())
                    }
                    vm.getBValueByDuplex = function() {
                        console.log('通过d2获得组件B的选项值: ' + avalon.vmodels.dropdown.d2)
                    }
                });
                avalon.scan();
            });
        </script>
        <style>
            .margin50 {
                margin-bottom: 50px;
            }
        </style>
    </head>
    <body>
        <div class="wrapper">
            <div ms-controller="dropdown" class="frame">
                <div class="margin50">
                    <h2>dropdownlist（下拉框）</h2>
                    <p>通过duplex的设置获取选中值或者通过调用getSelected方法来得到选中值</p>
                    <select ms-widget="dropdownlist, a">
                        <optgroup label="group1">
                            <option value="value1">Value 1</option>
                            <option>textValue</option>

                            <option value="value3">Value 3</option>
                        </optgroup>
                        <optgroup label="group2">
                            <option value="value4">Value 4</option>
                            <option value="value5" disabled>Value 5</option>
                            <option value="value6">Value 6</option>
                        </optgroup>
                    </select>
                </div>

                <div class="margin50">
                    <select ms-widget="dropdownlist, b" ms-duplex='d2'>
                        <optgroup label="group1">
                            <option value="value1">Value 1</option>
                            <option>textValue</option>
                            <option label="label" value="labelValue">with label</option>
                            <option value="value2">Value 2</option>
                            <option value="value3">Value 3</option>
                        </optgroup>

                        </optgroup>
                    </select>
                </div>
                <button ms-click='getAValue'>调用getSelected获取组件a的选项值</button>
                <button ms-click='getBValue'>调用getSelected获取组件b的选项值</button>
                <button ms-click='getBValueByDuplex'>通过duplex获取组件b的选项值</button>
            </div>
        </div>
