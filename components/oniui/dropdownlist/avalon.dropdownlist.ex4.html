dropdownlist（下拉框）

配置getRealTimeData根据搜索内容实时从远程获取数据来渲染选项列表，注意必须配置realTimeData为true

<!DOCTYPE html>
<html>
<head>
    <script src="../avalon.js"></script>
    <script type="text/javascript">
        function getData(search) {
            var data = [],
                random = Math.floor(Math.random() * 40 + 5)
            for (var i = 0; i < random; i++) {
                var value = search + Math.floor(Math.random() * 100)
                var item = {
                    value: value,
                    label: value
                }
                data.push(item)
            }
            return data
        }
        require(['dropdownlist/avalon.dropdownlist.js', "ready!"], function() {
            avalon.define('dropdown', function(vm) {
                vm.$skipArray = ['dropdownlist']
                vm.dropdownlist = {
                    getRealTimeData: function(search, dropdownlist) {
                        setTimeout(function() {
                            dropdownlist.render(getData(search))
                        }, 100)
                    },
                    realTimeData: true
                }
            });
            avalon.scan();
        });
    </script>
</head>
<body>
<div class="wrapper" ms-controller="dropdown">
    <div>
        <h2>dropdownlist（下拉框）</h2>
        <p>配置getRealTimeData根据搜索内容实时从远程获取数据来渲染选项列表，注意必须配置realTimeData为true</p>
        <select ms-widget="dropdownlist, a"> 
        </select>
    </div>
</div>
</body>
</html>
