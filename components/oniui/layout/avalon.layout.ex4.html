北区配置尺寸为{{northRegion.size}}。

东区配置尺寸为{{eastRegion.size}}。
直接改变Layout的大小时，各区域会按照百分比配置重新计算。但是经过拖放改变区域尺寸后，该区域配置的百分比失效并被实际尺寸覆盖。

                        <input ms-widget="textbox" ms-duplex-number="layoutWidth" type="textbox"/>
                        <input ms-widget="textbox" ms-duplex-number="layoutHeight" type="textbox" />
                        <div ms-css-width="layoutWidth" ms-css-height="layoutHeight" style="margin: 10px 0 0;overflow:hidden;">
                            <div ms-widget="layout,layoutVM,$layoutVMOptions" class="container">
                                <div ms-layout-dock="north">

                        require(["./layout/avalon.layout", "./button/avalon.button", "./textbox/avalon.textbox"], function() {
                            avalon.define("demo", function(vm) {
                                vm.$layoutVMOptions = {
                                    northRegion: {
                                        size: "20%",
                                        resizable: true,
                                        regionClass: "north"
                                    },
                                    eastRegion: {
                                        size: 120,
                                        resizable: true
                                    },
                                    westRegion: {
                                        size: "20%",
                                        resizable: true
                                    },
                                    stretchMax: true,
                                    resizerSize: 5,
                                    regionBorderWidth: 1
                                };
                                vm.layoutWidth = 600;
                                vm.layoutHeight = 500;
                                var resizeFn = function() {
                                    avalon.vmodels.layoutVM.updateLayoutSize(avalon.vmodels.layoutVM);
                                };
                                vm.$watch('layoutWidth', resizeFn);
                                vm.$watch('layoutHeight', resizeFn);
                            })
                            avalon.scan()
                        })
